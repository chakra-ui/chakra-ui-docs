---
id: spinner
category: feedback
title: Spinner
package: '@chakra-ui/spinner'
description:
  Spinners provide a visual cue that an action is processing awaiting a course
  of change or a result.
---

## Import

```js
import { Spinner } from '@chakra-ui/react'
```

## Usage

```jsx
import { Spinner } from '@chakra-ui/react'

export default function Example() {
  return <Spinner />
}
```

### Spinner with Color

```jsx
import { Spinner } from '@chakra-ui/react'

export default function Example() {
  return <Spinner color='red.500' />
}
```

### Spinner with different size

```jsx
import { Stack, Spinner } from '@chakra-ui/react'

export default function Example() {
  return (
    <Stack direction='row' spacing={4}>
      <Spinner size='xs' />
      <Spinner size='sm' />
      <Spinner size='md' />
      <Spinner size='lg' />
      <Spinner size='xl' />
    </Stack>
  )
}
```

### Spinner with empty area color

```jsx
import { Spinner } from '@chakra-ui/react'

export default function Example() {
  return (
    <Spinner
      thickness='4px'
      speed='0.65s'
      emptyColor='gray.200'
      color='blue.500'
      size='xl'
    />
  )
}
```
